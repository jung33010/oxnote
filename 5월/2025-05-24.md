# 초기값 설정시 무조건 반복문 위에 써라!!

def solution(n):
    y = n
    x = [n]
    while y != 1:
        if y % 2 == 0:
            y = y/2
            x.append(y)
        else:
            y =3*y+1
            x.append(y)
    return x
